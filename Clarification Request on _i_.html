python
import streamlit as st
import wikipedia

wikipedia.set_lang('en')

def get_answer(question):
    try:
        search_results = wikipedia.search(question)
        if not search_results:
            return "Sorry, no answer found."
        page = wikipedia.page(search_results[0])
        return page.summary[:500]
    except wikipedia.exceptions.DisambiguationError as e:
        return f"Your query is too broad. Options: {e.options}"
    except wikipedia.exceptions.PageError:
        return "Sorry, page not found."
    except Exception as e:
        return f"An error occurred: {str(e)}"

# Streamlit Frontend
st.title("Simple Chatbot (Wikipedia-powered)")

user_question = st.text_input("Ask me anything:")

if user_question:
    answer = get_answer(user_question)
    st.write("**Answer:**", answer)

  1. Run it with this command:

    streamlit run chatbot.py

  2. A beautiful chat page will open in your browser!

No paid services. No API keys. No OpenAI charges. It runs completely on your computer fetching answers from Wikipedia only.
